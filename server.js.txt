const express = require('express');
const mongoose = require('mongoose');
const bodyParser = require('body-parser');
const app = express();

// âœ… Middleware to parse form data
app.use(bodyParser.urlencoded({ extended: true }));

// âœ… Serve form directly (no need for index.html)
app.get('/', (req, res) => {
  res.send(`
    <form action="/submit" method="POST">
      <label>Name:</label><input name="name" /><br />
      <label>Email:</label><input name="email" /><br />
      <button type="submit">Submit</button>
    </form>
  `);
});

// âœ… MongoDB Connection
mongoose.connect('mongodb+srv://kavana:Chinnu123@cluster0.fsfshdm.mongodb.net/userData?retryWrites=true&w=majority&appName=Cluster0', {
  useNewUrlParser: true,
  useUnifiedTopology: true
}).then(() => {
  console.log('âœ… Connected to MongoDB Atlas!');
}).catch((error) => {
  console.error('âŒ MongoDB connection error:', error);
});

// âœ… Create Schema and Model
const userSchema = new mongoose.Schema({
  name: String,
  email: String
});
const User = mongoose.model('User', userSchema);

// âœ… Handle form submission
app.post('/submit', async (req, res) => {
  const user = new User(req.body);
  await user.save();
  res.send(`<h3>Thank you, ${user.name}!</h3><a href="/">Go back</a>`);
});

// âœ… Show all users
app.get('/users', async (req, res) => {
  const users = await User.find();
  res.json(users);
});

// âœ… Start server
app.listen(3000, () => {
  console.log('ðŸš€ Server running on http://localhost:3000');
});
